class Solution {
    func maxProfit(_ prices: [Int]) -> Int {
        var profitSum = 0
        for i in 1..<prices.endIndex {
            let profit = prices[i] - prices[i-1]
            if profit > 0 { 
                profitSum += profit
            }
        }
        return profitSum
    }
}
