class Solution {
    func threeSum(_ nums: [Int]) -> [[Int]] {
        let count = nums.count
        var ans = [[Int]]()
        for i in 0..<count-2 { 
            for j in i+1..<count-1 { 
                for k in j+1..<count { 
                    let a = nums[i]
                    let b = nums[j]
                    let c = nums[k]

                    if a+b+c == 0 {
                        ans.append([a,b,c])
                    }
                }
            }
        }
        return ans
    }
}
