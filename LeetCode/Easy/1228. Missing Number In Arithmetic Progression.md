class Solution {
    func missingNumber(_ arr: [Int]) -> Int {
        let count = arr.count
        let dif = (arr[count-1] - arr[0]) / count
        for i in 0..<count { 
            let predict = i * dif + arr[0] 
            if arr[i] != predict { 
                return predict
            }
        }
        return arr[0]
    }
}
