### O(n)
```swift
class Solution {
    func maxDistance(_ colors: [Int]) -> Int {
        let count = colors.count
        var left = 0, right = 1
        var distance = 0
        while left < right { 
            if colors[left] != colors[right] { 
                distance = max(distance, right-left)
            }
            right == count - 1 ? (left += 1) : (right += 1)
        }
        return distance
    }
}
```
