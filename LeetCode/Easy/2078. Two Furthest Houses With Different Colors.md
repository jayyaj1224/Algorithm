class Solution {
    func maxDistance(_ colors: [Int]) -> Int {
        let count = colors.count
        var maximum = 0
        for i in 0..<count { 
            for j in 1..<count { 
                if colors[i] != colors[j] { 
                    maximum = max(maximum, j-i)
                }
            }
        }
        return maximum
    }
}
