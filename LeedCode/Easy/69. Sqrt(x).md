class Solution {
    func mySqrt(_ x: Int) -> Int {
        guard x > 2 else { return Int(ceil(Double(x)/2)) }

        var start = 1, end = x

        while start<end { 
            let mid = start+(end-start)/2
            // print("\(mid) - \(start) - \(end)")
            if mid*mid<=x, (mid+1)*(mid+1)>x { 
                return mid
            } else if mid*mid>x { 
                end = mid
            } else { 
                start = mid
            }
        }
        return Int(end)
    }
}
