### #1 String Count
```Swift
class Solution {
    func hammingWeight(_ n: Int) -> Int {
        return String(n, radix: 2).filter { $0 == "1" }.count
    }
}
```

### #2 Bit by Bit
```Swift
class Solution {
    func hammingWeight(_ n: Int) -> Int {
        var num = n, count = 0

        while num != 0 { 
            if num&1 == 1 { count+=1 }
            num >>= 1
        }
        return count
    }
}
```
