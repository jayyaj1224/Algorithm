class Solution {
    func isAnagram(_ s: String, _ t: String) -> Bool {
        var t = Array(t)
        for (i,c) in s.enumerated() {
            if let _t = t.enumerated().first(where: {$0.element == c }) { 
                t.remove(at: _t.offset)
            } else { 
                return false
            }
        }
        return t.isEmpty
    }
}
